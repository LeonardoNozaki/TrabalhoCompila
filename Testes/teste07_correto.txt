//Funcao de multiplicar dois valores utilizandos dois meios diferentes
function multiplicar (mult1 : Int, mult2 : Int, tipo : Boolean) -> Int{
	if tipo == true{
		var i : Int;
		i = 1;
		var total : Int;
		total = 0;
		while i <= mult2{
			total = total + mult1;		
			i = i + 1;
		} 
		return total;
	}
	if tipo == false{
		var total : Int;
		total = mult1 * mult2;
		return total;
	}
}
